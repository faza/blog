1. Implement Event Registration Functionality

Step 1: Update Event Model

Add a relationship to track registrations:

# In app/models.py
from flask_login import UserMixin

class Event(db.Model):
    # ... existing fields ...
    attendees = db.relationship('User', secondary='registrations', backref=db.backref('attended_events', lazy='dynamic'))

class Registration(db.Model):
    __tablename__ = 'registrations'
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'), primary_key=True)
    event_id = db.Column(db.Integer, db.ForeignKey('event.id'), primary_key=True)
    registration_date = db.Column(db.DateTime, default=datetime.utcnow)

Step 2: Add Registration Route

Create a new route to handle event registration:

# In app/routes.py
from flask_login import login_required, current_user
from app.models import Event, Registration

@login_required
def register_for_event(event_id):
    event = Event.query.get_or_404(event_id)
    if current_user not in event.attendees:
        registration = Registration(user_id=current_user.id, event_id=event.id)
        db.session.add(registration)
        db.session.commit()
        flash('You have successfully registered for this event!', 'success')
    else:
        flash('You are already registered for this event.', 'info')
    return redirect(url_for('event_details', event_id=event_id))

Add a new route for the user dashboard:

# In app/routes.py
@login_required
def dashboard():
    created_events = Event.query.filter_by(organizer_id=current_user.id).all()
    registered_events = current_user.attended_events.all()
    return render_template('dashboard.html', created_events=created_events, registered_events=registered_events)

3. Add Search Functionality

Step 1: Create Search Route

Add a new route to handle search queries:

# In app/routes.py
from sqlalchemy import or_

@app.route('/search')
def search_events():
    query = request.args.get('query')
    if query:
        events = Event.query.filter(or_(
            Event.title.ilike(f'%{query}%'),
            Event.description.ilike(f'%{query}%'),
            Event.location.ilike(f'%{query}%')
        )).all()
    else:
        events = []
    return render_template('search_results.html', events=events, query=query)

Step 2: Create Search Form
